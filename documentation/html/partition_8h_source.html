412 inline void Partition<CgalMesh_EPICK>::determineBaseContours()
413 {
414 this->m_base_contours.reset();
415
416 std::vector<int> unordered_base_contour_edges;
417 std::vector<bool> edge_list;
418 edge_list.resize(this->m_mesh.number_of_halfedges());
419
420 for (const auto &f : this->m_mesh.faces())
421 {
422 const auto &e1 = this->m_mesh.halfedge(f);
423 const auto &e0 = this->m_mesh.prev(e1);
424 const auto &e2 = this->m_mesh.next(e1);
425
426 std::vector<CgalMesh_EPICK::Halfedge_index> edges;
427 if (this->m_base_plane.isLineOnPlane(Line(e1, this->m_mesh), 1e-2))
428 {
429 edges.emplace_back(e1);
430 }
431 if (this->m_base_plane.isLineOnPlane(Line(e0, this->m_mesh), 1e-2))
432 {
433 edges.emplace_back(e0);
434 }
435 if (this->m_base_plane.isLineOnPlane(Line(e2, this->m_mesh), 1e-2))
436 {
437 edges.emplace_back(e2);
438 }
439
440 if (edges.size() == 1)
441 {
442 unordered_base_contour_edges.emplace_back(edges[0].id());
443 edge_list[edges[0].id()] = true;
444 }
445 }
446
447 CgalMesh_EPICK::Halfedge_index source_he;
448 CgalMesh_EPICK::Vertex_index source_v;
449 CgalMesh_EPICK::Vertex_index end_v;
450 SO::Polygon contour;
451 contour.setPlane(this->m_base_plane);
452
453 while (unordered_base_contour_edges.size())
454 {
455 auto it = std::find(edge_list.begin(), edge_list.end(), true);
456 int index_to_start = std::distance(edge_list.begin(), it);
457 source_he = CgalMesh_EPICK::Halfedge_index(index_to_start);
458 source_v = this->m_mesh.source(source_he);
459 end_v = this->m_mesh.target(source_he);
460
461 const auto &cgal_point = this->m_mesh.point(source_v);
462 contour.addPointToBack(Eigen::Vector3d(cgal_point.x(), cgal_point.y(), cgal_point.z()));
463
464 auto it_source_he = std::find(unordered_base_contour_edges.begin(), unordered_base_contour_edges.end(), source_he.idx());
465 if (it_source_he != unordered_base_contour_edges.end())
466 {
467 edge_list[*it_source_he] = false;
468 unordered_base_contour_edges.erase(it_source_he);
469 }
470
471 while (end_v.idx() != source_v.idx())
472 {
473 for (const auto &he : this->m_mesh.halfedges_around_target(source_he))
474 {
475 auto it = std::find(unordered_base_contour_edges.begin(), unordered_base_contour_edges.end(), this->m_mesh.opposite(he).idx());
476 if (it != unordered_base_contour_edges.end())
477 {
478 source_he = this->m_mesh.opposite(he);
479 end_v = this->m_mesh.target(source_he);
480
481 const auto &cgal_point = this->m_mesh.point(this->m_mesh.source(source_he));
482 contour.addPointToBack(Eigen::Vector3d(cgal_point.x(), cgal_point.y(), cgal_point.z()));
483 edge_list[*it] = false;
484 unordered_base_contour_edges.erase(it);
485 break;
486 }
487 }
488 }
489 m_base_contours.addPolygon(contour);
490 contour.reset();
491 contour.setPlane(this->m_base_plane);
492 }
