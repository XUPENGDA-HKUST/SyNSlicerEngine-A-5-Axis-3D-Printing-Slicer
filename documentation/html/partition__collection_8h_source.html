158 {
159 std::ifstream inputFile(file_name); // Replace with your file path
160 int status = 0; // 0: Find ".stl'. 1: Find "Base". 2: Find "Slice".
161
162 if (inputFile.is_open()) {
163 std::string line;
164 SO::PrintingLayerCollection printing_layers;
165 SO::Plane prev_slicing_plane;
166 while (std::getline(inputFile, line))
167 {
168 if (status == 0)
169 {
170 std::string::size_type n;
171 n = line.find(".stl");
172 if (n != std::string::npos)
173 {
174 SO::Partition<CgalMesh_EPICK> partition(line);
175 m_partitions.emplace_back(partition);
176 status = 1;
177 }
178 else
179 {
180 if (this->m_partitions.size() > 0) { return true; }
181 else { return false; }
182 }
183 }
184 else if (status == 1)
185 {
186 std::string::size_type n;
187 n = line.find("Base");
188 if (n != std::string::npos)
189 {
190 char ss[10];
191 double a, b, c, d, e, f;
192 std::sscanf(line.c_str(), "%s %lf %lf %lf %lf %lf %lf", &ss, &a, &b, &c, &d, &e, &f);
193 SO::Plane base_plane(Eigen::Vector3d(a, b, c), Eigen::Vector3d(d, e, f));
194 m_partitions.back().setBasePlane(base_plane);
195 prev_slicing_plane = base_plane;
196 status = 2;
197 printing_layers = m_partitions.back().getPrintingLayers();
198 }
199 else
200 {
201 spdlog::error("Saved file not valid. Load fails.");
202 }
203 }
204 else if (status == 2)
205 {
206 std::string::size_type n;
207 n = line.find("Slice");
208 if (n != std::string::npos)
209 {
210 char ss[10];
211 double a, b, c, d, e, f;

213 SO::Plane slicing_plane(Eigen::Vector3d(a, b, c), Eigen::Vector3d(d, e, f));
214

221 n = line.find("End");

225 printing_layers = SO::PrintingLayerCollection();
226 status = 0;
