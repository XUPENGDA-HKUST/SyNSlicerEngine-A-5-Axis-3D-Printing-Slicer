426 inline bool Polyhedron<T>::fillHoles()
427 {
428 if (PMP::does_self_intersect(this->m_mesh))
429 {
430 std::cout << "Mesh is self-intersected, hole filling aborted!" << std::endl;
431 return false;
432 }
433
434 unsigned int nb_holes = 0;
435 std::vector<boost::graph_traits<CgalMesh_EPICK>::halfedge_descriptor> border_cycles;
436 PMP::extract_boundary_cycles(this->m_mesh, std::back_inserter(border_cycles));
437
438 for (boost::graph_traits<CgalMesh_EPICK>::halfedge_descriptor h : border_cycles)
439 {
440 std::vector<boost::graph_traits<CgalMesh_EPICK>::face_descriptor> patch_facets;
441 std::vector<boost::graph_traits<CgalMesh_EPICK>::vertex_descriptor> patch_vertices;
442 PMP::triangulate_and_refine_hole(this->m_mesh,
443 h,
444 std::back_inserter(patch_facets),
445 std::back_inserter(patch_vertices));
446 ++nb_holes;
447 }
448 return true;
449 }

458 inline bool Polyhedron<CgalMesh_EPICK>::getUnion(CgalMesh_EPICK mesh)
459 {
460 auto add_vertex_to_mesh = [](CgalPoint_EPICK &point, CgalMesh_EPICK &mesh) {
461 Eigen::Vector3d p0(point.x(), point.y(), point.z());
462 for (CgalMesh_EPICK::Vertex_index vid : mesh.vertices())
463 {
464 Eigen::Vector3d p1(mesh.point(vid).x(), mesh.point(vid).y(), mesh.point(vid).z());
465 if ((p0 - p1).norm() < 1e-6)
466 {
467 return vid;
468 };
469 }
470 CgalMesh_EPICK::Vertex_index a = mesh.add_vertex(point);
471 return a;
472 };
473
474 for (CgalMesh_EPICK::face_index face_idx : mesh.faces())
475 {
476 std::vector<CgalMesh_EPICK::vertex_index> vertex_list;
477 for (CgalMesh_EPICK::vertex_index vertex_idx : mesh.vertices_around_face(mesh.halfedge(face_idx)))
478 {
479 vertex_list.push_back(add_vertex_to_mesh(mesh.point(vertex_idx), this->m_mesh));
480 }
481 this->m_mesh.add_face(vertex_list[0], vertex_list[1], vertex_list[2]);
482 }
483 return false;
